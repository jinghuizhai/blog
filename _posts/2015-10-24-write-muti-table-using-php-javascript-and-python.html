用javascript,php和python写乘法口诀

PHP: 

function mutil_table($max){
	for($i = 1;$i<$max;$i++){
			$arr = array();
			for($j = 1;$j<$i+1;$j++){
				$arr[] = $i."*".$j."=".$i*$j;
			}
			echo join($arr,',')."<br>";
		}
	}
调用：mutil_table(9);
